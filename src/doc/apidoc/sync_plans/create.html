POST /api/organizations/:organization_id/sync_plans
Create a sync plan

Examples

POST /api/organizations/org_2d4c613/sync_plans/
{
  "sync_plan": {
    "name": "plan_1",
    "interval": "weekly",
    "description": "plan 1 desc.",
    "sync_date": "2012-06-06T09:00:00+02:00"
  }
}

  "name": "plan_1",

  "description": "plan 1 desc.",
  "sync_date": "2012-06-06T07:00:00Z",
  "created_at": "2012-06-05T07:04:07Z"
}
