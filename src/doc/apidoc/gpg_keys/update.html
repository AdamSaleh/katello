PUT /api/gpg_keys/:id

Examples

PUT /api/gpg_keys/18/
{
  "gpg_key": {
    "name": "Renamed-Dummy-Package-Generator"
  }
}

  "name": "Renamed-Dummy-Package-Generator",
  "content": "gpg create from input                   [ \u001b[32mOK\u001b[m\u000f ]\n",
  "organization_id": 28,
  "id": 18,
  "created_at": "2012-06-05T06:58:10Z"
}
